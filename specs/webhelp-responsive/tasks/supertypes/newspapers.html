          - b
        # Serial

          - s

      - - 008[21]
        - n
    - AND:
      # Serial control
      - - 006[0]
        - s
      # Newspaper
      - - 006[4]
        - n
    - OR:
      - lower-case(760t) contains 'newspaper'
